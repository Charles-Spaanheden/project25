get('/cards') do
    db = databas("db/databas.db")
    result = db.execute("SELECT * FROM card")
    admin = admin(session[:id])
    logged_in = true
    if session[:id] == nil
        logged_in = false
    end
    slim(:"cards/index",locals:{cards:result,admin:admin,logged_in:logged_in}) 

end
